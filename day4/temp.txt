import { useState } from "react";
import ChildComp from "./components/child.component";

let App = () => {
  let [show, setShow] = useState(true);

  return (
    <div>
      <h1>Hooks Custom</h1>
      <button onClick={()=> setShow(!show)}>Show / Hide</button>
      { show ? <ChildComp/> : "Component is hidden"}
    </div>
  );
}

export default App;
